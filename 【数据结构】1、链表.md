* 剑指6、从尾到头打印链表
方法一：递归
```python
class Solution:
    # 返回从尾部到头部的列表值序列，例如[1,2,3]
    def printListFromTailToHead(self, listNode):
        # write code here
        self.head = listNode
        cur = self.head
        print(self.travel(self,cur))

    def travel(self,cur):
        if not cur:
            return []
        return self.printListFromTailToHead(cur.next) + [cur.val]
```


方法三，从头到尾读，然后然后进行数组翻转
```python
    def travel(self):
        l = []

            return []

            cur = self.head
 ```

```python
        # 空链表或者只有一个元素的链表
        if head == None or head.next == None:
            return head
        new_head = head
        while head != None and head.next != None:
 ```

 ```python
class Solution(object):
    def swapPairs(self, head: ListNode) -> ListNode:
        """
        :type head: ListNode
        :rtype: ListNode
        """

        # If the list has no node or has only one node left.
        if not head or not head.next:
            return head

        # Nodes to be swapped
        first_node = head
        second_node = head.next

        # Swapping
        first_node.next  = self.swapPairs(second_node.next)
        second_node.next = first_node

        # Now the head is the second node
        return second_node
 ```

* LeetCode 234、回文链表（快慢指针+反转链表,空间复杂度O(1)）
一开始想着用栈解决，以为是LeetCode 20. 有效的括号 的思路，后来发现不对，如果是奇数个，则中间那个数字只出现一次，所以该思路并不好
但是可以放进栈一半，反转之后与链表后一半比较
 ``` python
# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def isPalindrome(self, head: ListNode) -> bool:
        # 如果只有一个或者没有元素，返回true
        if not head or not head.next:
            return True
        # 找到链表的中间位置
        mid = self.get_mid(head)
        second_half_end = mid.next
        # 反转后半段
        second_start = self.reverse_list(second_half_end)
        second_half_start = second_start
        # 分别从head和反转后的head开始遍历，如果不等说明False
        result = True
        while result and second_start is not None:
            if head.val != second_start.val:
                result = False
            head = head.next
            second_start = second_start.next
        # 将反转部分恢复
        mid.next = self.reverse_list(second_half_start)
        return result

    # 快慢指针
    def get_mid(self,head):
        fast,slow = head,head
        while fast.next and fast.next.next:
            fast = fast.next.next
            slow = slow.next
        return slow
    
    def reverse_list(self,head):
        pre = None
        cur = head
        while cur:
            temp = cur.next
            cur.next = pre
            pre = cur
            cur = temp
        return pre
  ```  
* LeetCode 725、分隔链表
思路：双指针，i用来寻找首先出现的值大于等于x的节点，j用来寻找首先出现的不满足条件位置的值小于x的节点，交换位置，时间复杂度O(n), 空间复杂度O(1)。
```python
class Solution:
    def partition(self, head: ListNode, x: int) -> ListNode:
        i, j = head, head
        while i and j :
            if i.val < x:
                i = i.next 
            else:
                if j == head:
                    if i.next:
                        j = i.next #第一次从i后面的节点开始寻找值小于x的节点
                    else:
                        return head
                while j is not None and j.val >= x: #找到则交换i，j节点的值, i,j指针后移
                    j = j.next 
                if j is not None:
                    i.val, j.val = j.val, i.val 
                    i = i.next
        return head
  ```

```python  
class Solution:

        if not head:
            return head
  ```  
